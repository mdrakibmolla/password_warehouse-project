from string import *
from itertools import product

count = 0

value = ascii_letters + digits + punctuation

for i in range(3, 4):
    for j in product(value, repeat=i):
        word = "".join(j)
        file = open("password_warehouse.txt", "a")
        try:
            file.write(word)
            file.write("\n")
            count += 1
            print(f" {count} passwords are Generated :- {word}")
        except Exception as e:
            print(e)
